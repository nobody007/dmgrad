# Function for Drawing Gray-Scale Image
draw.image <- function(x, index) {
  img <- x[index,,]
  img.array <- t(img) %>% array_reshape(c(img_rows, img_cols, 1, 1))
  plot(cimg(img.array), axes = FALSE)  
}

# Function for History Plots
history.plot <- function(history) {
  if (is.data.frame(history)) { gtb <- history }
  else { gtb <- as.data.frame(history$metrics) }
  minim <- which.min(gtb$val_loss)
  epoch <- 1:nrow(gtb)
  g <- ggplot(gtb) + theme_test() + 
    theme(plot.title = element_text(hjust = 0.5), legend.position = 'bottom') +
    scale_color_manual('set', values = c(training = 'darkblue', validation = 'red1'))
  g1 <- g + 
    geom_vline(aes(xintercept = minim), color = 'gray', linetype = 2) +
    geom_line(aes(epoch, loss, color = 'training')) +
    geom_line(aes(epoch, val_loss, color = 'validation')) +
    labs(title = 'Loss')
  g2 <- g +
    geom_vline(aes(xintercept = minim), color = 'gray', linetype = 2) +
    geom_line(aes(epoch, acc, color = 'training')) +
    geom_line(aes(epoch, val_acc, color = 'validation')) +
    labs(title = 'Accuracy')
  gridExtra::grid.arrange(g1, g2, ncol = 2)
}

Load Dataset

# Input image dimensions
img_rows <- 28
img_cols <- 28

# Number of classes
num_classes <- 10
mnist <- dataset_mnist()
x_train <- mnist$train$x
y_train <- mnist$train$y
x_test <- mnist$test$x
y_test <- mnist$test$y
  • Sample Images for 10 Classes
for (class in 1:num_classes) {
  id <- sample(which(y_train == class -1), 5)
  par(mfrow = c(1, 5), mar = c(1, 1, 1, 1))
  for (i in id) draw.image(x_train, i)
}

Data Transformation

# Redefine dimension of train/test inputs (with a channel)
x_train <- array_reshape(x_train, c(nrow(x_train), img_rows, img_cols, 1))
x_test <- array_reshape(x_test, c(nrow(x_test), img_rows, img_cols, 1))
input_shape <- c(img_rows, img_cols, 1)

# Transform RGB values into [0, 1] range
x_train <- x_train / 255
x_test <- x_test / 255

cat('x_train_shape:', dim(x_train), '\n')
## x_train_shape: 60000 28 28 1
cat(nrow(x_train), 'train samples\n')
## 60000 train samples
cat(nrow(x_test), 'test samples\n')
## 10000 test samples
# Convert class vectors to binary class matrices
y_train <- to_categorical(y_train, num_classes)
y_test <- to_categorical(y_test, num_classes)

Training

  • Define Model
model <- keras_model_sequential() %>%
  layer_conv_2d(filters = 32, kernel_size = c(3, 3), activation = 'relu',
                input_shape = input_shape) %>% 
  layer_conv_2d(filters = 64, kernel_size = c(3, 3), activation = 'relu') %>% 
  layer_max_pooling_2d(pool_size = c(2, 2)) %>% 
  layer_dropout(rate = 0.25) %>% 
  layer_flatten() %>% 
  layer_dense(units = 128, activation = 'relu') %>% 
  layer_dropout(rate = 0.5) %>% 
  layer_dense(units = num_classes, activation = 'softmax')
  • Compile model
model %>% compile(
  loss = 'categorical_crossentropy',
  optimizer = optimizer_adam(),
  metrics = c('accuracy')
)

  • Train model
batch_size <- 128
epochs <- 12

history <- model %>% fit(
  x_train, y_train, validation_split = 0.2,
  epochs = epochs, batch_size = batch_size,
  verbose = 2
)

# Save Model Object & History
write.csv(as.data.frame(history$metrics), 'history-example.csv', row.names = F)
save_model_hdf5(model, 'model-example.hdf5')
