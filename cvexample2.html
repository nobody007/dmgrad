#hard to see which one is better

 

#compute 2 fold CV

 

p<-0.5

n<-nrow(Auto)

 

rep.n<-100

mse<-rep(0,rep.n)

for (i in 1:rep.n){

    train.ind<-sample(n,as.integer(n*p))

    train.data<-Auto[train.ind,]

    test.data<-Auto[-train.ind,]

 

    lm1<-lm(mpg~horsepower,data=train.data)

    lm1.pred<-predict(lm1,test.data)

    mse[i]<-mean((lm1.pred-test.data$mpg)^2)

}

 

rep.n<-100

mse<-matrix(0,ncol=10,nrow=rep.n)

for (i in 1:rep.n){

    train.ind<-sample(n,as.integer(n*p))

    train.data<-Auto[train.ind,]

    test.data<-Auto[-train.ind,]

 

    for (j in 1:10){

        lm1<-lm(mpg~poly(horsepower,j,raw=T),data=train.data)

        lm1.pred<-predict(lm1,test.data)

        mse[i,j]<-mean((lm1.pred-test.data$mpg)^2)

    }

}

 

 

par(mfrow=c(1,1))

plot(colMeans(mse),type="b",xlab="degree of poly", ylab="MSE")

 

plot(mse[1,],type="l")

for (j in 2:10){

    lines(mse[j,],col=j)

}

 

 

plot(mse[1,],type="l",ylim=c(15,30))

for (j in 2:10){

    lines(mse[j,],col=j)

}

 

plot(mse[1,],type="l",ylim=c(15,30))

for (j in 2:100){

    lines(mse[j,],col=j)

}
