# example in page 24-25. curse of dimensionality

# x unif [-1,1]^p, p : dimension of x

# true function y=f(x)=exp(-8*||x||^2)

# use 1-nn method to estimate y at x=0

 

#generate x from dimension 1 - 10

 

 

myrand<-function(a,b,p,n){

#generate p-dim r.v from unif(a,b)

#n is sample size

    xx<-matrix(runif(n*p,a,b),nrow=n,ncol=p)

    return(xx)

}

 

nn<-1000

sim.n<-1000

sim.res<-matrix(0,nrow=10,ncol=4)

for (p in 1:10){

temp1<-matrix(0,ncol=sim.n,nrow=2)

    for ( i in 1:sim.n){

        #generate r.v's

        x<-myrand(-1,1,p,nn)

        dist1<-rowSums(x^2)

        min1<-min(dist1)

        y1<-exp(-8*min1)

        avgdist1<-sqrt(min1)

        temp1[1,i]<-y1

        temp1[2,i]<-avgdist1

    }

    sim.res[p,1]<-mean(temp1[1,]) # estimate at x=0

    sim.res[p,2]<-mean(temp1[2,]) # avg.distance to nearest neighbor to 0

    sim.res[p,3]<-mean((temp1[1,]-1)^2) # sq.bias

    sim.res[p,4]<-var(temp1[1,]) # variance

 

}

 

par(mfrow=c(2,2))

mydim<-1:10

plot(mydim,sim.res[,1],type="b",xlab="dim",ylab="",main="avg. estimate at x=0, true=1")

plot(mydim,sim.res[,2],type="b",xlab="dim",ylab="",main="avg. distance to nn at x=0")

plot(mydim,sim.res[,3],type="b",xlab="dim",ylab="MSE",main="MSE vs. dimension")

lines(mydim,sim.res[,4],type="b",col=2)

lines(mydim,sim.res[,3]+sim.res[,4],type="b",col=3)

legend(1,0.95,c("sq.bias","var","MSE"),lty=rep(1,3),col=1:3)
